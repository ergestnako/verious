List

Class

          
<div class="verious">
  <ul>
    <li>
      ...
    </li>
  </ul>
</div>

Prefix

          
<ul class="vs-ul">
  <li class="vs-li">
    ...
  </li>
</ul>

<ul class="vs-ul--unstyled">
  <li class="vs-li">
    ...
  </li>
</ul>

Mixin

          
ul {
  @include vs-ul();

  &.unstyled {
    @include vs-ul--unstyled();
  }
}

li {
  @include vs-li(/* $size */);
}
